🚀 Quick Start

# Extract
tar -xzf channelos-0.1.0-alpha.tar.gz
cd channelos

# Read guides
cat README.md
cat IMPLEMENTATION_GUIDE.md

# Setup
./scripts/setup.sh esp32s3

# Build (after adding implementations)
make TARGET=esp32s3
